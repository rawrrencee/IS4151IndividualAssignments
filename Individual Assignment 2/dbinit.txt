CREATE TABLE `events` (`recordId` INTEGER PRIMARY KEY AUTOINCREMENT, `deviceId` INTEGER NOT NULL, `deviceName` CHARACTER(5) NOT NULL, `district` INTEGER NOT NULL, `event` INTEGER NOT NULL, `timestamp` DATETIME NOT NULL, `toCloud` BOOLEAN DEFAULT 0 NOT NULL);

CREATE TABLE `trackers` (`recordId` INTEGER PRIMARY KEY AUTOINCREMENT, `deviceId` INTEGER NOT NULL, `deviceName` CHARACTER(5) NOT NULL, `district` INTEGER NOT NULL, `temperature` INTEGER NOT NULL, `fever` CHARACTER(1) NOT NULL, `intruder` CHARACTER(1) NOT NULL, `timestamp` DATETIME NOT NULL, `toCloud` BOOLEAN DEFAULT 0 NOT NULL);
